核心程式碼（OpenClaw 技能中的 web-control.js）：

const { chromium } = require('playwright');

async function controlWebPage(url, task) {
    // 1. 打開瀏覽器（學生可以看到操作過程）
    const browser = await chromium.launch({ headless: false });
    const page = await browser.newPage();
    await page.goto(url);

    // 2. 注入 Page Agent SDK
    await page.addScriptTag({
        url: 'https://cdn.jsdelivr.net/npm/page-agent@1.5.7/dist/iife/page-agent.demo.js'
    });

    // 3. 用自然語言執行任務
    await page.waitForTimeout(2000);
    const result = await page.evaluate(async (task) => {
        const agent = new PageAgent({ language: 'zh-TW' });
        return await agent.execute(task);
    }, task);

    await browser.close();
    return result;
}
